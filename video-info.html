<script>

document.getElementById('videoInput').addEventListener('change', function (e) {

  const file = e.target.files[0];

  if (!file) return;


  const resultDiv = document.getElementById('result');

  resultDiv.innerHTML = "<p>⏳ 正在加载视频信息...</p>";


  const video = document.createElement('video');

  video.preload = 'metadata';

  video.muted = true;

  video.playsInline = true; // iPhone 必须加

  video.src = URL.createObjectURL(file);


  // 必须添加到 DOM 才能触发 loadedmetadata

  video.style.display = 'none';

  document.body.appendChild(video);


  video.addEventListener('loadedmetadata', function () {

    const width = video.videoWidth;

    const height = video.videoHeight;

    const duration = video.duration.toFixed(2);

    const sizeMB = (file.size / (1024 * 1024)).toFixed(2);


    // 显示在页面

    resultDiv.innerHTML = `

      <p>🎞 文件名：<b>${file.name}</b></p>

      <p>📐 分辨率：<b>${width} × ${height}</b></p>

      <p>🕒 时长：<b>${duration} 秒</b></p>

      <p>💾 大小：<b>${sizeMB} MB</b></p>

    `;


    // ✅ 打印到控制台

    console.log("视频信息：", {

      name: file.name,

      width: width,

      height: height,

      duration: duration,

      sizeMB: sizeMB

    });


    URL.revokeObjectURL(video.src);

    video.remove();

  });


  video.addEventListener('error', function () {

    resultDiv.innerHTML = `<p style="color:red;">⚠️ 无法读取视频元数据，请尝试其他文件。</p>`;

  });

});

</script>
